120 void finalize() {
121 // Fuente
122 font_obj = reserve_object();
123 add_object(font_obj, "<< /Type /Font /Subtype /Type1 /BaseFont /Helvetica >>");
124
125 // Objetos de contenido y página
126 for (const auto& content : page_contents) {
127 int content_obj = reserve_object();
128 int page_obj = reserve_object();
129 content_objs.push_back(content_obj);
130 page_objs.push_back(page_obj);
131
132 std::string content_stream = "<< /Length " + std::to_string(content.size()) +
133 " >>\nstream\n" + content + "\nendstream";
134 add_object(content_obj, content_stream);
135
136 std::string page_content = "<< /Type /Page /Parent " + std::to_string(pages_obj) +
137 " 0 R /Resources << /Font << /F1 " + std::to_string(font_obj) +
138 " 0 R >> >> /MediaBox [0 0 612 792] /Contents " +
139 std::to_string(content_obj) + " 0 R >>";
140 add_object(page_obj, page_content);
141 }
142
143 // Objeto de páginas
144 std::string kids;
145 for (int page_obj : page_objs)
146 kids += std::to_string(page_obj) + " 0 R ";
147 pages_obj = reserve_object();
148 std::string pages_content = "<< /Type /Pages /Kids [" + kids + "] /Count " +
149 std::to_string(page_objs.size()) + " >>";
150 add_object(pages_obj, pages_content);
151
152 // Catálogo
153 catalog_obj = reserve_object();
154 std::string catalog_content = "<< /Type /Catalog /Pages " + std::to_string(pages_obj) + " 0 R >>";
155 add_object(catalog_obj, catalog_content);
156
157 // XREF y trailer
158 size_t xref_offset = file.tellp();
159 file << "xref\n0 " << current_object_number << "\n";
160 file << "0000000000 65535 f \n";
161 for (int i = 1; i < current_object_number; ++i) {
162 file << std::setfill('0') << std::setw(10) << offsets[i] << " 00000 n \n";
163 }
164 file << "trailer\n<< /Size " << current_object_number << " /Root " << catalog_obj <<
165 " 0 R >>\nstartxref\n" << xref_offset << "\n%%EOF\n";
166 }

257 void createPDF(const std::string& txtFile, const std::string& pdfFile) {
258 std::ifstream inFile(txtFile);
259 if (!inFile.is_open()) {
260 std::cerr << "Error: No se pudo abrir el archivo " << txtFile << std::endl;
261 return;
262 }
263
264 std::vector<std::vector<std::string>> tablaDatos;
265 std::string line;
266 while (std::getline(inFile, line)) {
267 std::vector<std::string> campos;
268 size_t start = 0, end;
269 while ((end = line.find('|', start)) != std::string::npos) {
270 campos.push_back(line.substr(start, end - start));
271 start = end + 1;
272 }
273 campos.push_back(line.substr(start));
274 if (!campos.empty()) {
275 tablaDatos.push_back(campos);
276 }
277 }
278 inFile.close();
279
280 PDFCreator pdf(pdfFile);
281
282 float x = 50, y = 800, font_size = 11, line_height = 16;
283 bool titulo_impreso = false;
284
285 pdf.start_page(); // INICIA LA PRIMERA PÁGINA
286
287 for (const auto& fila : tablaDatos) {
288 // Salto de página si es necesario
289 if (y < 120) {
290 pdf.end_page(); // Termina la página actual
291 pdf.start_page(); // Inicia una nueva página
292 y = 800;
293 titulo_impreso = false;
294 }
295
296 if (!titulo_impreso) {
297 pdf.add_text("BANCO", x, y, 18);
298 y -= line_height * 2;
299 pdf.add_text("Datos personales de clientes", x, y, 14);
300 y -= line_height * 2;
301 titulo_impreso = true;
302 }
303
304 if (fila.size() < 12) continue; // Salta si faltan campos
305
306 pdf.add_text("----------------------------------------", x, y, font_size);
307 y -= line_height;
308
309 std::string nombre = fila[1] + " " + fila[2];
310 std::string apellido = fila[3] + " " + fila[4];
311 pdf.add_text("Nombre: " + nombre, x, y, font_size);
312 y -= line_height;
313 pdf.add_text("Apellido: " + apellido, x, y, font_size);
314 y -= line_height;
315 pdf.add_text("Telefono: " + fila[5], x, y, font_size);
316 y -= line_height;
317 pdf.add_text("Correo: " + fila[6], x, y, font_size);
318 y -= line_height;
319 pdf.add_text("Usuario: " + fila[7], x, y, font_size);
320 y -= line_height;
321 pdf.add_text("Cuenta: " + fila[9], x, y, font_size);
322 y -= line_height;
323 pdf.add_text("Saldo: " + fila[10], x, y, font_size);
324 y -= line_height;
325 pdf.add_text("Fecha Nac.: " + fila[11], x, y, font_size);
326 y -= line_height;
327 pdf.add_text("----------------------------------------", x, y, font_size);
328 y -= line_height * 2;
329 }
330
331 pdf.end_page(); // Termina la última página
332 pdf.finalize();
333 std::cout << "PDF creado exitosamente: " << pdfFile << std::endl;
334}
