48 auto leftrotate = [](uint32 x, uint32 c) { return (x << c) | (x >> (32 - c)); };
49
50 // Inicializar variables
51 uint32 a0 = 0x67452301, b0 = 0xefcdab89, c0 = 0x98badcfe, d0 = 0x10325476;
52
53 // Preprocesamiento
54 vector<unsigned char> msg(str.begin(), str.end());
55 size_t orig_len_bits = msg.size() * 8;
56 msg.push_back(0x80);
57 while ((msg.size() * 8) % 512 != 448) msg.push_back(0);
58 for (int i = 0; i < 8; ++i) msg.push_back((orig_len_bits >> (8 * i)) & 0xFF);
59
60 // Procesar en bloques de 512 bits
61 for (size_t offset = 0; offset < msg.size(); offset += 64) {
62 uint32 M[16];
63 for (int i = 0; i < 16; ++i) {
64 M[i] = 0;
65 for (int j = 0; j < 4; ++j)
66 M[i] |= msg[offset + i * 4 + j] << (j * 8);
67 }
68 uint32 A = a0, B = b0, C = c0, D = d0;
69 for (int i = 0; i < 64; ++i) {
70 uint32 F, g;
71 if (i < 16) { F = (B & C) | (~B & D); g = i; }
72 else if (i < 32) { F = (D & B) | (~D & C); g = (5 * i + 1) % 16; }
73 else if (i < 48) { F = B ^ C ^ D; g = (3 * i + 5) % 16; }
74 else { F = C ^ (B | ~D); g = (7 * i) % 16; }
75 uint32 temp = D;
76 D = C;
77 C = B;
78 B = B + leftrotate(A + F + K[i] + M[g], s[(i / 16) * 4 + (i % 4)]);
79 A = temp;
80 }
81 a0 += A; b0 += B; c0 += C; d0 += D;
82 }
83 stringstream ss;
84 for (uint32 v : {a0, b0, c0, d0})
85 for (int i = 0; i < 4; ++i)

88 // --- FIN IMPLEMENTACIÓN MD5 ---

107class TablaHash {
108 // Usamos ListaDoble para manejar colisiones
109 vector<ListaDobleCircular<Hash>*> table;
110 // Capacidad de la tabla hash
111 int capacidad;
112 int hashFunction(const string& key) {
113 int hash = 0;
114 for (char c : key) hash = (hash * 31 + c) % capacidad;
115 return hash;
116 }
117 public:
118 TablaHash(int cap) : capacidad(cap), table(cap, new ListaDobleCircular<Hash>()) {}
119 ~TablaHash() {
120 for (auto* lista : table) {
121 delete lista;
122 }
123 }
124 // Inserta un nuevo elemento o actualiza el valor si ya existe
125 void insert(const string& key, int value) {
126 int idx = hashFunction(key);
127 auto* lista = table[idx];
128 auto* nodo = lista->buscar([&](Hash* e){ return e->getOriginal() == key; });
129 if (nodo) {
130 nodo->setValue(value);
131 } else {
132 lista->insertar(new Hash(key, value));
133 }
134 }
135 bool get(const string& key, int& value) {
136 int idx = hashFunction(key);
137 auto* lista = table[idx];
138 auto* nodo = lista->buscar([&](Hash* e){ return e->getOriginal() == key; });
139 if (nodo) {
140 value = nodo->getValue();
141 return true;
142 }
143 return false;
144 }
145 void remove(const string& key) {
146 int idx = hashFunction(key);
147 auto* lista = table[idx];
148 auto* nodo = lista->buscar([&](Hash* e){ return e->getOriginal() == key; });
149 if (nodo) {
150 lista->eliminar(nodo);
151 }
152 }
